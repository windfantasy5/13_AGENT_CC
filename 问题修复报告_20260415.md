### 1. 添加LLM流式调用方法

**文件：** `backend/app/services/llm_service.py`

```python
async def chat_completion_stream(
    self,
    messages: List[Dict[str, str]],
    temperature: float = 0.7,
    max_tokens: int = 2000
):
    """流式聊天补全"""
    balancer = await get_llm_balancer()
    current_model = await balancer.get_current_model()

    try:
        if current_model == LLMBalancer.PRIMARY_MODEL:
            async for chunk in self._stream_openai(...):
                yield chunk
        else:
            async for chunk in self._stream_ollama(...):
                yield chunk

        await balancer.record_success(current_model)
    except Exception as e:
        logger.error(f"LLM流式调用失败: {e}")
        await balancer.record_failure(current_model)
        raise

async def _stream_openai(self, messages, model, temperature, max_tokens):
    """OpenAI流式调用"""
    stream = await self.openai_client.chat.completions.create(
        model=model,
        messages=messages,
        temperature=temperature,
        max_tokens=max_tokens,
        stream=True
    )

    async for chunk in stream:
        if chunk.choices[0].delta.content:
            yield {
                "content": chunk.choices[0].delta.content,
                "model": model
            }

async def _stream_ollama(self, messages, model, temperature, max_tokens):
    """Ollama流式调用"""
    stream = await self.ollama_client.chat(
        model=model,
        messages=messages,
        options={"temperature": temperature, "num_predict": max_tokens},
        stream=True
    )

    async for chunk in stream:
        if chunk.get("message", {}).get("content"):
            yield {
                "content": chunk["message"]["content"],
                "model": model
            }
```

```
用户提问
    ↓

    ↓

    ↓

    ↓

2. RAG检索 → 发送 rag_context
3. LLM流式调用 → 发送 assistant_chunk (多次)
```
